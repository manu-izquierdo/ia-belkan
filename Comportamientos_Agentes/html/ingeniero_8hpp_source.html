40// Comparador para la priority_queue:
41// La prioridad mas alta es la (longitud, impacto) mas pequeña, asi que devolvemos true cuando
42// 'a' debe ir DESPUES que 'b' (ya que la priority_queue ordena de mayor a menor).
43struct ComparaTuberia {
44 bool operator()(const NodoTuberia &a, const NodoTuberia &b) const {
45 // Lex: primero longitud, luego impacto
46 if (a.longitud != b.longitud)
47 return a.longitud > b.longitud;
48 return a.impacto > b.impacto;
49 }
50};
